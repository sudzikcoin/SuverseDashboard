"use client";
import { useState } from "react";
import Link from "next/link";
import { usePathname } from "next/navigation";
import { Menu, X } from "lucide-react";

const navItems = [
  { name: "Dashboard", href: "/admin" },
  { name: "Accountants", href: "/admin/accountants" },
  { name: "Companies", href: "/admin/companies" },
  { name: "Inventory", href: "/admin/inventory" },
  { name: "Purchase Orders", href: "/admin/orders" },
  { name: "Audit Log", href: "/admin/audit" },
];

export default function Sidebar() {
  const [open, setOpen] = useState(false);
  const pathname = usePathname();

  return (
    <>
      {/* Top bar (always visible on mobile) */}
      <div className="flex items-center justify-between md:hidden p-4 border-b border-white/10 bg-[#0a0f1a] z-40">
        <span className="text-lg font-semibold text-emerald-400">SuVerse</span>
        <button
          onClick={() => setOpen(!open)}
          className="p-2 focus:outline-none text-white"
        >
          {open ? <X size={24} /> : <Menu size={24} />}
        </button>
      </div>

      {/* Overlay (dark background when menu open) */}
      {open && (
        <div
          className="fixed inset-0 bg-black/40 backdrop-blur-sm z-40"
          onClick={() => setOpen(false)}
        />
      )}

      {/* Sidebar panel */}
      <div
        className={`fixed md:static top-0 left-0 h-full w-64 bg-[#0a0f1a] border-r border-white/10 text-white transform transition-transform duration-300 z-50 ${
          open ? "translate-x-0" : "-translate-x-full md:translate-x-0"
        }`}
      >
        <div className="flex flex-col h-full p-4 space-y-3">
          <h1 className="text-xl font-bold text-emerald-400 hidden md:block">
            SuVerse
          </h1>
          <p className="text-xs opacity-60 hidden md:block mb-4">
            Tax Credit Dashboard
          </p>

          {navItems.map((item) => (
            <Link
              key={item.href}
              href={item.href}
              className={`block rounded-md px-3 py-2 text-sm font-medium transition-colors ${
                pathname === item.href
                  ? "bg-emerald-600 text-black"
                  : "hover:bg-white/10 text-white/80"
              }`}
              onClick={() => setOpen(false)}
            >
              {item.name}
            </Link>
          ))}

          <div className="mt-auto pt-6 border-t border-white/10">
            <button
              onClick={() => (window.location.href = "/logout")}
              className="text-red-400 hover:text-red-500 text-sm"
            >
              Sign Out
            </button>
          </div>
        </div>
      </div>
    </>
  );
}
