export const COMPLETED_ORDER_STATUSES = ['COMPLETED', 'DEMO_COMPLETED'] as const;

export async function getCompanyPurchaseStats(companyId: string) {
  const orders = await prisma.order.findMany({
    where: {
      companyId,
      status: { in: COMPLETED_ORDER_STATUSES },
    },
    orderBy: { createdAt: 'desc' },
  });

  const totalCount = orders.length;
  const totalValue = orders.reduce(
    (sum, o) => sum + (o.faceValueUsd ?? 0),
    0,
  );

  return { orders, totalCount, totalValue };
}
