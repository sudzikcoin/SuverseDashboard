"use client";
import { useState } from "react";

type Props = {
  title?: string;
  companyId?: string; // for accountant uploading on behalf of a company
  defaultType?: string;
};

export default function FileUpload({ title = "Upload Document", companyId, defaultType = "KYC" }: Props) {
  const [file, setFile] = useState<File | null>(null);
  const [type, setType] = useState(defaultType);
  const [notes, setNotes] = useState("");
  const [busy, setBusy] = useState(false);
  const [msg, setMsg] = useState<string | null>(null);

  async function onSubmit(e: React.FormEvent) {
    e.preventDefault();
    if (!file) return;
    setBusy(true);
    setMsg(null);

    const fd = new FormData();
    fd.append("file", file);
    fd.append("type", type);
    if (companyId) fd.append("companyId", companyId);
    if (notes) fd.append("notes", notes);

    const res = await fetch("/api/upload", { method: "POST", body: fd });
    const j = await res.json();
    setBusy(false);
    if (!res.ok) {
      setMsg(`Error: ${j.error ? JSON.stringify(j.error) : "Upload failed"}`);
    } else {
      setMsg("Uploaded âœ“");
      setFile(null);
      setNotes("");
    }
  }

  return (
    <form onSubmit={onSubmit} className="rounded-2xl border border-white/10 bg-white/5 p-4 backdrop-blur">
      <h3 className="text-lg font-semibold mb-3">{title}</h3>
      <div className="grid gap-3">
        <input
          type="file"
          onChange={(e) => setFile(e.target.files?.[0] ?? null)}
          className="block w-full text-sm file:mr-4 file:rounded-full file:border-0 file:bg-emerald-600 file:px-4 file:py-2 file:text-white hover:file:bg-emerald-500"
        />
        <div className="flex gap-2">
          <input
            value={type}
            onChange={(e) => setType(e.target.value)}
            placeholder="Type (e.g., W9, KYC, POA)"
            className="flex-1 rounded-xl bg-black/20 px-3 py-2 outline-none border border-white/10"
          />
          <input
            value={notes}
            onChange={(e) => setNotes(e.target.value)}
            placeholder="Notes (optional)"
            className="flex-1 rounded-xl bg-black/20 px-3 py-2 outline-none border border-white/10"
          />
        </div>
        <button
          disabled={busy || !file}
          className="rounded-xl bg-emerald-600 px-4 py-2 text-white disabled:opacity-50"
        >
          {busy ? "Uploading..." : "Upload"}
        </button>
        {msg && <p className="text-sm text-white/80">{msg}</p>}
      </div>
    </form>
  );
}
