import { prisma } from '@/lib/prisma';
import { hash } from 'bcryptjs'; // or whatever you use

export async function ensureDemoBrokerUser() {
  const email = 'broker.demo@suverse.io';

  let user = await prisma.user.findUnique({ where: { email } });

  if (!user) {
    const broker = await prisma.broker.create({
      data: {
        name: 'Demo Broker',
        legalName: 'Demo Broker LLC',
        email,
      },
    });

    const passwordHash = await hash('demoBroker123', 10);

    user = await prisma.user.create({
      data: {
        email,
        name: 'Demo Broker',
        role: 'BROKER',
        brokerId: broker.id,
        passwordHash, // adapt to your schema (password, hashedPassword, etc.)
      },
    });
  } else if (user.role !== 'BROKER') {
    user = await prisma.user.update({
      where: { id: user.id },
      data: { role: 'BROKER' },
    });
  }

  return user;
}
