// app/api/auth/[...nextauth]/route.ts
import NextAuth from "next-auth";
import Credentials from "next-auth/providers/credentials";
import { PrismaClient } from "@prisma/client";
import { compare } from "bcryptjs"; // if you hash passwords
import { authOptionsBase } from "@/lib/auth-options-base"; // if you have base options; otherwise ignore
const prisma = new PrismaClient();

export const authOptions = {
  // ... your existing providers, pages, etc.
  providers: [
    // keep existing providers
    Credentials({
      name: "Credentials",
      credentials: {
        email: { label: "Email", type: "text" },
        password: { label: "Password", type: "password" },
      },
      async authorize(credentials) {
        if (!credentials?.email || !credentials?.password) return null;
        const user = await prisma.user.findUnique({
          where: { email: credentials.email },
        });
        if (!user) return null;

        // if you use hashed passwords
        // const ok = await compare(credentials.password, user.passwordHash);
        // if (!ok) return null;

        // if you store plain (dev), compare directly:
        if (user.password !== credentials.password) return null;

        return { id: String(user.id), email: user.email, name: user.name || null };
      },
    }),
  ],

  callbacks: {
    async session({ session, token }) {
      // Ensure we have a user id from token or session
      const email = session.user?.email ?? null;

      // Try to find a company linked to this user.
      // NOTE: adjust the relation if your schema differs.
      // Option A: Company has userId owner
      let company = null;
      if (email) {
        company = await prisma.company.findFirst({
          where: {
            OR: [
              { ownerEmail: email },      // if you keep ownerEmail
              { contactEmail: email },    // optional fallback
              { users: { some: { email } } }, // if M:N users-companies
            ],
          },
          select: { legalName: true },
        });
      }

      session.user.companyName = company?.legalName ?? null;
      return session;
    },

    async jwt({ token, user }) {
      // keep your existing logic; not required to store company in token
      if (user?.id) token.userId = user.id;
      return token;
    },
  },
};

const handler = NextAuth(authOptions);
export { handler as GET, handler as POST };
